// onBeatHit is called every beat
function onBeatHit
    var beat, 0
    set beat, getCurrentBeat
    
    var beatMod, 0
    set beatMod, mod beat 8
    
    if beatMod == 7
        playCharAnim bf, hey, true
    endif
end function
