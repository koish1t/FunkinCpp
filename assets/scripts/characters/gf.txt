// GF Character Script

// onCreate is called when the PlayState is created
function onCreate
    // basically checks if dad is gf and if he is the game moves dad to gf's pos before hiding og gf
    var player2, ""
    set player2, getPlayer2
    
    if player2 == "gf"
        // gets geef's pos
        var gfX, 0.0
        var gfY, 0.0
        set gfX, getCharX gf
        set gfY, getCharY gf
        
        setCharPos dad, gfX, gfY        
        setCharVisible gf, false
    endif
end function
